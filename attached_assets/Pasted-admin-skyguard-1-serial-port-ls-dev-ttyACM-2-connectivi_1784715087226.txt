admin@skyguard:~ $ # 1. Провери дали се е появил serial port
ls /dev/ttyACM*

# 2. Свали connectivity HCI firmware за nRF52840
cd ~
wget https://github.com/NordicSemiconductor/pc-ble-driver/releases/download/v4.1.4/connectivity_4.1.4_usb_with_s140_6.1.1.zip

# 3. Флашни
~/nrfutil-env/bin/nrfutil dfu usb-serial \
  -pkg connectivity_4.1.4_usb_with_s140_6.1.1.zip \
  -p /dev/ttyACM0 \
  -b 115200

# 4. Изчакай 3 сек и провери
sleep 3
lsusb | grep -i "1915\|nordic"
hciconfig -a
